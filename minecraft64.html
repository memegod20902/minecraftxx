import java.awt.*; import java.awt.event.KeyAdapter; import java.awt.event.KeyEvent; import java.awt.event.MouseAdapter; import java.awt.event.MouseEvent; import java.util.ArrayList; public class MinecraftLikeGame extends JPanel { // Game Constants static final int BLOCK_SIZE = 50; static final int PLAYER_SIZE = 50; static final int WINDOW_WIDTH = 800; static final int WINDOW_HEIGHT = 600; // Player and world properties int playerX = WINDOW_WIDTH / 2; int playerY = WINDOW_HEIGHT / 2; int playerSpeed = 5; // World representation (list of blocks) ArrayList world = new ArrayList<>(); public MinecraftLikeGame() { // Initialize world with ground blocks for (int x = 0; x < WINDOW_WIDTH; x += BLOCK_SIZE) { for (int y = WINDOW_HEIGHT - 100; y < WINDOW_HEIGHT; y += BLOCK_SIZE) { world.add(new Block(x, y, Color.BROWN)); // Ground blocks } } // KeyListener for player movement addKeyListener(new KeyAdapter() { @Override public void keyPressed(KeyEvent e) { if (e.getKeyCode() == KeyEvent.VK_LEFT) { playerX -= playerSpeed; } else if (e.getKeyCode() == KeyEvent.VK_RIGHT) { playerX += playerSpeed; } else if (e.getKeyCode() == KeyEvent.VK_UP) { playerY -= playerSpeed; } else if (e.getKeyCode() == KeyEvent.VK_DOWN) { playerY += playerSpeed; } // Prevent player from going out of bounds playerX = Math.max(0, Math.min(WINDOW_WIDTH - PLAYER_SIZE, playerX)); playerY = Math.max(0, Math.min(WINDOW_HEIGHT - PLAYER_SIZE, playerY)); repaint(); } }); // MouseListener for block placement/removal addMouseListener(new MouseAdapter() { @Override public void mouseClicked(MouseEvent e) { int mouseX = e.getX(); int mouseY = e.getY(); // Find which block is clicked for (Block block : world) { if (block.isClicked(mouseX, mouseY)) { block.toggle(); // Toggle block color (removal/addition) break; } } repaint(); } }); setFocusable(true); } // Method to draw the world and player @Override protected void paintComponent(Graphics g) { super.paintComponent(g); // Draw the world (blocks) for (Block block : world) { block.draw(g); } // Draw the player g.setColor(Color.GREEN); // Player color g.fillRect(playerX, playerY, PLAYER_SIZE, PLAYER_SIZE); } // Block class to represent each block class Block { int x, y; Color color; Block(int x, int y, Color color) { this.x = x; this.y = y; this.color = color; } void draw(Graphics g) { g.setColor(color); g.fillRect(x, y, BLOCK_SIZE, BLOCK_SIZE); } boolean isClicked(int mouseX, int mouseY) { return mouseX >= x && mouseX <= x + BLOCK_SIZE && mouseY >= y && mouseY <= y + BLOCK_SIZE; } void toggle() { // Toggle between sky (blue) and ground (brown) color = (color == Color.BROWN) ? Color.CYAN : Color.BROWN; } } // Main method to run the game public static void main(String[] args) { JFrame frame = new JFrame("Minecraft-like Game"); MinecraftLikeGame gamePanel = new MinecraftLikeGame(); import static org.lwjgl.system.MemoryUtil.*; public class MinecraftLike3D { private long window; private int width = 800, height = 600; // Camera and movement private float xPos = 0, yPos = 1, zPos = 5; if (window == NULL) { throw new RuntimeException("Failed to create the GLFW window"); } glfwMakeContextCurrent(window); glfwSwapInterval(1); glfwShowWindow(window); // Initialize OpenGL GL.createCapabilities(); glEnable(GL_DEPTH_TEST); } public void loop() { while (!glfwWindowShouldClose(window)) { glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT); // Update camera updateCamera(); // Draw simple 3D cube world (for demonstration) drawCube(-1, -1, -5, 1);
